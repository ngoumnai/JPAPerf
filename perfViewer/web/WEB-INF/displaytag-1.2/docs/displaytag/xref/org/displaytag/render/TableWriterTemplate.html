36  /**
37   * A template that encapsulates and drives the construction of a table based on a given table model and configuration.
38   * This class is meant to be extended by classes that build tables sharing the same structure, sorting, and grouping,
39   * but that write them in different formats and to various destinations. Subclasses must provide the format- and
40   * destination-specific implementations of the abstract methods this class calls to build a table. (Background: This
41   * class came about because our users wanted to export tables to Excel and PDF just as they were presented in HTML. It
42   * originates with the TableTagData.writeHTMLData method, factoring its logic so that it can be re-used by classes that
43   * write the tables as PDF, Excel, RTF and other formats. TableTagData.writeHTMLData now calls an HTML extension of this
44   * class to write tables in HTML format to a JSP page.)
45   * @author Jorge L. Barroso
46   * @version $Id$
47   */

74      /**
75       * Given a table model, this method creates a table, sorting and grouping it per its configuration, while delegating
76       * where and how it writes the table to subclass objects. (Background: This method refactors
77       * TableTagData.writeHTMLData method. See above.)
78       * @param model The table model used to build the table.
79       * @param id This table's page id.
80       * @throws JspException if any exception thrown while constructing the tablei, it is caught and rethrown as a
81       * JspException. Extension classes may throw all sorts of exceptions, depending on their respective formats and
82       * destinations.
83       */
84      public void writeTable(TableModel model, String id) throws JspException
85      {
86          try
87          {
88              // table id used for logging
89              this.id = id;
90  
91              TableProperties properties = model.getProperties();
92  
93              if (log.isDebugEnabled())
94              {
95                  log.debug("[" + this.id + "] writeTable called for table [" + this.id + "]");
96              }
97  
98              // Handle empty table
99              boolean noItems = model.getRowListPage().size() == 0;
100             if (noItems && !properties.getEmptyListShowTable())
101             {
102                 writeEmptyListMessage(properties.getEmptyListMessage());
103                 return;
104             }
105 
106             // Put the page stuff there if it needs to be there...
107             if (properties.getAddPagingBannerTop())
108             {
109                 // search result and navigation bar
110                 writeTopBanner(model);
111             }
112 
113             // open table
114             writeTableOpener(model);
115 
116             // render caption
117             if (model.getCaption() != null)
118             {
119                 writeCaption(model);
120             }
121 
122             // render headers
123             if (model.getProperties().getShowHeader())
124             {
125                 writeTableHeader(model);
126             }
127 
128             // render footer prior to body
129             if (model.getFooter() != null)
130             {
131                 writePreBodyFooter(model);
132             }
133 
134             // open table body
135             writeTableBodyOpener(model);
136 
137             // render table body
138             writeTableBody(model);
139 
140             // close table body
141             writeTableBodyCloser(model);
142 
143             // render footer after body
144             if (model.getFooter() != null)
145             {
146                 writePostBodyFooter(model);
147             }
148 
149             // close table
150             writeTableCloser(model);
151 
152             if (model.getTableDecorator() != null)
153             {
154                 writeDecoratedTableFinish(model);
155             }
156 
157             writeBottomBanner(model);
158 
159             if (log.isDebugEnabled())
160             {
161                 log.debug("[" + this.id + "] writeTable end");
162             }
163         }
164         catch (Exception e)
165         {
166             throw new JspException(e);
167         }
168     }

257     /**
258      * Given a table model, writes the table body content, sorting and grouping it per its configuration, while
259      * delegating where and how it writes to subclass objects. (Background: This method refactors
260      * TableTagData.writeTableBody method. See above.)
261      * @param model The table model used to build the table body.
262      * @throws Exception if an error is encountered while writing the table body.
263      */
264     private void writeTableBody(TableModel model) throws Exception
265     {
266         // Ok, start bouncing through our list (only the visible part)
267         RowIterator rowIterator = model.getRowIterator(false);
268 
269         // iterator on rows
270         TableDecorator tableDecorator = model.getTableDecorator();
271         Row previousRow = null;
272         Row currentRow = null;
273         Row nextRow = null;
274         Map previousRowValues = new HashMap(10);
275         Map currentRowValues = new HashMap(10);
276         Map nextRowValues = new HashMap(10);
277 
278         while (nextRow != null || rowIterator.hasNext())
279         {
280             // The first pass
281             if (currentRow == null)
282             {
283                 currentRow = rowIterator.next();
284             }
285             else
286             {
287                 previousRow = currentRow;
288                 currentRow = nextRow;
289             }
290 
291             if (previousRow != null)
292             {
293                 previousRowValues.putAll(currentRowValues);
294             }
295             if (!nextRowValues.isEmpty())
296             {
297                 currentRowValues.putAll(nextRowValues);
298             }
299             // handle the first pass
300             else
301             {
302                 ColumnIterator columnIterator = currentRow.getColumnIterator(model.getHeaderCellList());
303                 // iterator on columns
304                 if (log.isDebugEnabled())
305                 {
306                     log.debug(" creating ColumnIterator on " + model.getHeaderCellList());
307                 }
308                 while (columnIterator.hasNext())
309                 {
310                     Column column = columnIterator.nextColumn();
311 
312                     // Get the value to be displayed for the column
313                     column.initialize();
314                     CellStruct struct = new CellStruct(column, column.getChoppedAndLinkedValue());
315                     currentRowValues.put(new Integer(column.getHeaderCell().getColumnNumber()), struct);
316                 }
317             }
318 
319             nextRowValues.clear();
320             // Populate the next row values
321             nextRow = rowIterator.hasNext() ? rowIterator.next() : null;
322             if (nextRow != null)
323             {
324                 ColumnIterator columnIterator = nextRow.getColumnIterator(model.getHeaderCellList());
325                 // iterator on columns
326                 if (log.isDebugEnabled())
327                 {
328                     log.debug(" creating ColumnIterator on " + model.getHeaderCellList());
329                 }
330                 while (columnIterator.hasNext())
331                 {
332                     Column column = columnIterator.nextColumn();
333                     column.initialize();
334                     // Get the value to be displayed for the column
335                     CellStruct struct = new CellStruct(column, column.getChoppedAndLinkedValue());
336                     nextRowValues.put(new Integer(column.getHeaderCell().getColumnNumber()), struct);
337                 }
338             }
339             // now we are going to create the current row; reset the decorator to the current row
340             if (tableDecorator != null)
341             {
342                 tableDecorator.initRow(currentRow.getObject(), currentRow.getRowNumber(), currentRow.getRowNumber()
343                     + rowIterator.getPageOffset());
344             }
345 
346             Iterator headerCellsIter = model.getHeaderCellList().iterator();
347             ArrayList structsForRow = new ArrayList(model.getHeaderCellList().size());
348             lowestEndedGroup = NO_RESET_GROUP;
349             lowestStartedGroup = NO_RESET_GROUP;
350             while (headerCellsIter.hasNext())
351             {
352                 HeaderCell header = (HeaderCell) headerCellsIter.next();
353 
354                 // Get the value to be displayed for the column
355                 CellStruct struct = (CellStruct) currentRowValues.get(new Integer(header.getColumnNumber()));
356                 struct.decoratedValue = struct.bodyValue;
357                 // Check and see if there is a grouping transition. If there is, then notify the decorator
358                 if (header.getGroup() != -1)
359                 {
360                     CellStruct prior = (CellStruct) previousRowValues.get(new Integer(header.getColumnNumber()));
361                     CellStruct next = (CellStruct) nextRowValues.get(new Integer(header.getColumnNumber()));
362                     // Why npe?
363                     String priorBodyValue = prior != null ? prior.bodyValue : null;
364                     String nextBodyValue = next != null ? next.bodyValue : null;
365                     short groupingValue = groupColumns(struct.bodyValue,
366                             priorBodyValue, nextBodyValue, header.getGroup());
367 
368                     if (tableDecorator != null)
369                     {
370                         switch (groupingValue)
371                         {
372                             case GROUP_START :
373                                 tableDecorator.startOfGroup(struct.bodyValue, header.getGroup());
374                                 break;
375                             case GROUP_END :
376                                 tableDecorator.endOfGroup(struct.bodyValue, header.getGroup());
377                                 break;
378                             case GROUP_START_AND_END :
379                                 tableDecorator.startOfGroup(struct.bodyValue, header.getGroup());
380                                 tableDecorator.endOfGroup(struct.bodyValue, header.getGroup());
381                                 break;
382                             default :
383                                 break;
384                         }
385                     }
386                     if (tableDecorator != null)
387                     {
388                         struct.decoratedValue = tableDecorator.displayGroupedValue(struct.bodyValue,
389                                 groupingValue, header.getColumnNumber());
390                     }
391                     else if (groupingValue == GROUP_END || groupingValue == GROUP_NO_CHANGE)
392                     {
393                         struct.decoratedValue = TagConstants.EMPTY_STRING;
394                     }
395                 }
396                 structsForRow.add(struct);
397             }
398 
399             if (tableDecorator != null)
400             {
401                 writeDecoratedRowStart(model);
402             }
403             // open row
404             writeRowOpener(currentRow);
405 
406             for (Iterator iterator = structsForRow.iterator(); iterator.hasNext();)
407             {
408                 CellStruct struct = (CellStruct) iterator.next();
409                 writeColumnOpener(struct.column);
410                 writeColumnValue(struct.decoratedValue, struct.column);
411                 writeColumnCloser(struct.column);
412             }
413 
414             if (model.isEmpty())
415             {
416                 if (log.isDebugEnabled())
417                 {
418                     log.debug("[" + this.id + "] table has no columns");
419                 }
420                 // render empty row
421                 writeRowWithNoColumns(currentRow.getObject().toString());
422             }
423 
424             // close row
425             writeRowCloser(currentRow);
426             // decorate row finish
427             if (model.getTableDecorator() != null)
428             {
429                 writeDecoratedRowFinish(model);
430             }
431         }
432 
433         // render empty list message
434         if (model.getRowListPage().size() == 0)
435         {
436             writeEmptyListRowMessage(MessageFormat.format(
437                 model.getProperties().getEmptyListRowMessage(),
438                 new Object[]{new Integer(model.getNumberOfColumns())}));
439         }
440     }

510     /**
511      * This takes a column value and grouping index as the argument. It then groups the column and returns the
512      * appropriate string back to the caller.
513      * @param value String current cell value
514      * @return String
515      */
516     protected short groupColumns(String value, String previous, String next, int currentGroup)
517     {
518 
519         short groupingKey = GROUP_NO_CHANGE;
520         if (lowestEndedGroup < currentGroup)
521         {
522             // if a lower group has ended, cascade so that all subgroups end as well
523             groupingKey += GROUP_END;
524         }
525         else if (next == null || !ObjectUtils.equals(value, next))
526         {
527             // at the end of the list
528             groupingKey += GROUP_END;
529             lowestEndedGroup = currentGroup;
530         }
531 
532         if (lowestStartedGroup < currentGroup)
533         {
534             // if a lower group has started, cascade so that all subgroups restart as well
535             groupingKey += GROUP_START;
536         }
537         else if (previous == null || !ObjectUtils.equals(value, previous))
538         {
539             // At the start of the list
540             groupingKey += GROUP_START;
541             lowestStartedGroup = currentGroup;
542         }
543         return groupingKey;
544     }
