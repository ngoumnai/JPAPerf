130     private void _initialize(Servlet servlet,
131                  ServletRequest request,
132                  ServletResponse response,
133                  String errorPageURL,
134                  boolean needsSession,
135                  int bufferSize,
136                  boolean autoFlush) throws IOException {
137 
138     // initialize state
139     this.servlet = servlet;
140     this.config = servlet.getServletConfig();
141     this.context = config.getServletContext();
142     this.needsSession = needsSession;
143     this.errorPageURL = errorPageURL;
144     this.bufferSize = bufferSize;
145     this.autoFlush = autoFlush;
146     this.request = request;
147     this.response = response;
148 
149     // Setup session (if required)
150     if (request instanceof HttpServletRequest && needsSession)
151         this.session = ((HttpServletRequest)request).getSession();
152     if (needsSession && session == null)
153         throw new IllegalStateException
154                 ("Page needs a session and none is available");
155 
156         // initialize the initial out ...
157         depth = -1;
158         if (this.baseOut == null) {
159             this.baseOut = new JspWriterImpl(response, bufferSize, autoFlush);
160         } else {
161             this.baseOut.init(response, bufferSize, autoFlush);
162         }
163         this.out = baseOut;
164 
165     // register names/values as per spec
166     setAttribute(OUT, this.out);
167     setAttribute(REQUEST, request);
168     setAttribute(RESPONSE, response);
169 
170     if (session != null)
171         setAttribute(SESSION, session);
172 
173     setAttribute(PAGE, servlet);
174     setAttribute(CONFIG, config);
175     setAttribute(PAGECONTEXT, this);
176     setAttribute(APPLICATION, context);
177 
178     isIncluded = request.getAttribute(
179         "javax.servlet.include.servlet_path") != null;
180     }

182     public void release() {
183         out = baseOut;
184     try {
185         if (isIncluded) {
186         ((JspWriterImpl)out).flushBuffer();
187                 // push it into the including jspWriter
188         } else {
189                 // Old code:
190                 //out.flush();
191                 // Do not flush the buffer even if we're not included (i.e.
192                 // we are the main page. The servlet will flush it and close
193                 // the stream.
194                 ((JspWriterImpl)out).flushBuffer();
195             }
196     } catch (IOException ex) {
197         log.warn("Internal error flushing the buffer in release()");
198     }
199 
200     servlet = null;
201     config = null;
202     context = null;
203     needsSession = false;
204     errorPageURL = null;
205     bufferSize = JspWriter.DEFAULT_BUFFER;
206     autoFlush = true;
207     request = null;
208     response = null;
209         depth = -1;
210     baseOut.recycle();
211     session = null;
212 
213     attributes.clear();
214     }

259         switch (scope) {
260             case PAGE_SCOPE:
261             return attributes.get(name);
262 
263             case REQUEST_SCOPE:
264             return request.getAttribute(name);
265 
266             case SESSION_SCOPE:
267             if (session == null) {
268                 throw new IllegalStateException(
269                         Localizer.getMessage("jsp.error.page.noSession"));
270             }
271             return session.getAttribute(name);
272 
273             case APPLICATION_SCOPE:
274             return context.getAttribute(name);
275 
276             default:
277             throw new IllegalArgumentException("Invalid scope");
278         }

422     private int doGetAttributeScope(String name){
423         if (attributes.get(name) != null)
424             return PAGE_SCOPE;
425 
426         if (request.getAttribute(name) != null)
427             return REQUEST_SCOPE;
428 
429         if (session != null) {
430             if (session.getAttribute(name) != null)
431                 return SESSION_SCOPE;
432         }

435             return APPLICATION_SCOPE;
436 
437         return 0;

667     private void doForward(String relativeUrlPath)
668         throws ServletException, IOException{
669 
670         // JSP.4.5 If the buffer was flushed, throw IllegalStateException
671         try {
672             out.clear();
673         } catch (IOException ex) {
674             IllegalStateException ise =
675                 new IllegalStateException(Localizer.getMessage(
676                             "jsp.error.attempt_to_clear_flushed_buffer"));
677             ise.initCause(ex);
678             throw ise;
679         }
680 
681         // Make sure that the response object is not the wrapper for include
682         while (response instanceof ServletResponseWrapperInclude) {
683             response = ((ServletResponseWrapperInclude)response).getResponse();
684         }
685 
686         final String path = getAbsolutePathRelativeToContext(relativeUrlPath);
687         String includeUri
688             = (String) request.getAttribute(Constants.INC_SERVLET_PATH);
689 
690         final ServletResponse fresponse = response;
691         final ServletRequest frequest = request;
692 
693         if (includeUri != null)
694             request.removeAttribute(Constants.INC_SERVLET_PATH);
695         try {
696             context.getRequestDispatcher(path).forward(request, response);
697         } finally {
698             if (includeUri != null)
699                 request.setAttribute(Constants.INC_SERVLET_PATH, includeUri);
700             request.setAttribute(Constants.FORWARD_SEEN, "true");
701         }
702     }

708     public JspWriter pushBody(Writer writer) {
709         depth++;
710         if (depth >= outs.length) {
711             BodyContentImpl[] newOuts = new BodyContentImpl[depth + 1];
712             for (int i=0; i<outs.length; i++) {
713                 newOuts[i] = outs[i];
714             }
715             newOuts[depth] = new BodyContentImpl(out);
716             outs = newOuts;
717         }
718 
719     outs[depth].setWriter(writer);
720         out = outs[depth];
721 
722     // Update the value of the "out" attribute in the page scope
723     // attribute namespace of this PageContext
724     setAttribute(OUT, out);
725 
726         return outs[depth];
727     }

729     public JspWriter popBody() {
730         depth--;
731         if (depth >= 0) {
732             out = outs[depth];
733         } else {
734             out = baseOut;
735         }
736 
737     // Update the value of the "out" attribute in the page scope
738     // attribute namespace of this PageContext
739     setAttribute(OUT, out);
740 
741         return out;
742     }

761     public void handlePageException(final Throwable t)
762         throws IOException, ServletException
763     {
764         if (t == null)
765             throw new NullPointerException("null Throwable");
766 
767         if (SecurityUtil.isPackageProtectionEnabled()){
768             try{
769                 AccessController.doPrivileged(new PrivilegedExceptionAction(){
770                     public Object run() throws Exception{
771                         doHandlePageException(t);
772                         return null;
773                     }
774                 });
775             } catch (PrivilegedActionException e){
776                 Exception ex =  e.getException();
777                 if (ex instanceof IOException){
778                     throw (IOException)ex;
779                 } else {
780                     throw (ServletException)ex;
781                 }
782             }
783         } else {
784             doHandlePageException(t);
785         }
786

789     private void doHandlePageException(Throwable t)
790         throws IOException, ServletException {
791 
792         if (errorPageURL != null && !errorPageURL.equals("")) {
793 
794             /*
795              * Set request attributes.
796              * Do not set the javax.servlet.error.exception attribute here
797              * (instead, set in the generated servlet code for the error page)
798              * in order to prevent the ErrorReportValve, which is invoked as
799              * part of forwarding the request to the error page, from
800              * throwing it if the response has not been committed (the response
801              * will have been committed if the error page is a JSP page).
802              */
803             request.setAttribute("javax.servlet.jsp.jspException", t);
804             request.setAttribute("javax.servlet.error.status_code",
805                 new Integer(HttpServletResponse.SC_INTERNAL_SERVER_ERROR));
806             request.setAttribute("javax.servlet.error.request_uri",
807             ((HttpServletRequest) request).getRequestURI());
808             request.setAttribute("javax.servlet.error.servlet_name",
809                      config.getServletName());
810             try {
811                 forward(errorPageURL);
812             } catch (IllegalStateException ise) {
813                 include(errorPageURL);
814             }
815 
816             // The error page could be inside an include.
817 
818             Object newException = request.getAttribute("javax.servlet.error.exception");
819 
820             // t==null means the attribute was not set.
821             if( (newException!= null) && (newException==t) ) {
822                 request.removeAttribute("javax.servlet.error.exception");
823             }
824 
825             // now clear the error code - to prevent double handling.
826             request.removeAttribute("javax.servlet.error.status_code");
827             request.removeAttribute("javax.servlet.error.request_uri");
828             request.removeAttribute("javax.servlet.error.status_code");
829             request.removeAttribute("javax.servlet.jsp.jspException");

831         } else {
832             // Otherwise throw the exception wrapped inside a ServletException.
833             // Set the exception as the root cause in the ServletException
834             // to get a stack trace for the real problem
835             if (t instanceof IOException) throw (IOException)t;
836             if (t instanceof ServletException) throw (ServletException)t;
837             if (t instanceof RuntimeException) throw (RuntimeException)t;
838 
839             Throwable rootCause = null;
840             if (t instanceof JspException) {
841                 rootCause = ((JspException) t).getRootCause();
842             } else if (t instanceof ELException) {
843                 rootCause = ((ELException) t).getRootCause();
844             }
845 
846             if (rootCause != null) {
847                 // FG start patch
848                 if (rootCause instanceof RuntimeException)
849                 {
850                     throw (RuntimeException) rootCause;
851                 }
852                 rootCause.printStackTrace();
853                 // FG end patch
854                 throw new ServletException(t.getClass().getName() + ": " +
855                                            t.getMessage(), rootCause);
856             }
857 
858             throw new ServletException(t);
859         }

869     private static String XmlEscape(String s) {
870         if (s == null) return null;
871         StringBuffer sb = new StringBuffer();
872         for(int i = 0; i < s.length(); i++) {
873             char c = s.charAt(i);
874             if (c == '<') {
875                 sb.append("&lt;");
876             } else if (c == '>') {
877                 sb.append("&gt;");
878             } else if (c == '\'') {
879                 sb.append("&#039;");    // &apos;
880             } else if (c == '&') {
881                 sb.append("&amp;");
882             } else if (c == '"') {
883                 sb.append("&#034;");    // &quot;
884             } else {
885                 sb.append(c);
886             }
887         }
888         return sb.toString();
889     }
