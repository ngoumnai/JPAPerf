54      /**
55       * Constructs a ServletUnitHttpRequest from a WebRequest object.
56       **/
57      ServletUnitHttpRequest( ServletMetaData servletRequest, WebRequest request, ServletUnitContext context, Dictionary clientHeaders, byte[] messageBody ) throws MalformedURLException {
58          if (context == null) throw new IllegalArgumentException( "Context must not be null" );
59  
60          _servletRequest = servletRequest;
61          _request = request;
62          _context = context;
63          _headers = new WebClient.HeaderDictionary();
64          _headers.addEntries( clientHeaders );
65          _headers.addEntries( request.getHeaders() );
66          setCookiesFromHeader( _headers );
67          _messageBody = messageBody;
68          _secure = request.getURL().getProtocol().equalsIgnoreCase( "https" );
69  
70          String contentTypeHeader = (String) _headers.get( "Content-Type" );
71          if (contentTypeHeader != null) {
72              String[] res = HttpUnitUtils.parseContentTypeHeader( contentTypeHeader );
73              _contentType = res[0];
74              _charset     = res[1];
75          }
76  
77          _requestContext = new RequestContext( request.getURL(), _charset  );
78  
79          if (_headers.get( "Content-Length") == null) _headers.put( "Content-Length", Integer.toString( messageBody.length ) );
80  
81          if (_messageBody != null && (_contentType == null || _contentType.indexOf( "x-www-form-urlencoded" ) >= 0 )) {
82              _requestContext.setMessageBody( _messageBody );
83          }
84      }

521     /**
522      * Parses the accept-language header to obtain a vector of preferred locales
523      * @return the preferred locales, sorted by qvalue
524      */
525     private Vector getPreferredLocales() {
526         if (_locales == null) {
527             _locales = new Vector();
528             String languages = getHeader( "accept-language" );
529             if (languages == null) {
530                 _locales.add( Locale.getDefault() );
531             } else {
532                  StringTokenizer st = new StringTokenizer( languages, "," );
533                  ArrayList al = new ArrayList();
534                  while (st.hasMoreTokens()) {
535                      String token = st.nextToken();
536                      al.add( new PrioritizedLocale( token ) );
537                  }
538                  Collections.sort( al );
539                  for (Iterator iterator = al.iterator(); iterator.hasNext();) {
540                      _locales.add( ((PrioritizedLocale) iterator.next()).getLocale() );
541                  }
542             }
543          }
544         return _locales;
545     }

557     /**
558      * Returns a RequestDispatcher object that acts as a wrapper for the resource located at the given path.
559      * A RequestDispatcher object can be used to forward a request to the resource or to include the
560      * resource in a response. The resource can be dynamic or static.
561      *
562      * The pathname specified may be relative, although it cannot extend outside the current servlet
563      * context. If the path begins with a "/" it is interpreted as relative to the current context root.
564      * This method returns null if the servlet container cannot return a RequestDispatcher.
565      *
566      * The difference between this method and ServletContext.getRequestDispatcher(java.lang.String)
567      * is that this method can take a relative path.
568      **/
569     public RequestDispatcher getRequestDispatcher( String path ) {
570         try {
571             if (!path.startsWith( "/" )) path = combinedPath( getServletPath(), path );
572             return _servletRequest.getServlet().getServletConfig().getServletContext().getRequestDispatcher( path );
573         } catch (ServletException e) {
574             return null;
575         }
576     }

579     private String combinedPath( String basePath, String relativePath ) {
580         if (basePath.indexOf( '/' ) < 0) return relativePath;
581         return basePath.substring( 0, basePath.lastIndexOf( '/' ) ) + '/' + relativePath;
582     }

662     /**
663      * Reconstructs the URL the client used to make the request.
664      * The returned URL contains a protocol, server name, port number, and server path, but
665      * it does not include query string parameters.
666      *
667      * Because this method returns a StringBuffer, not a string, you can modify the URL easily, for example,
668      * to append query parameters.
669      *
670      * This method is useful for creating redirect messages and for reporting errors.
671      *
672      * @since 1.3
673      */
674     public StringBuffer getRequestURL() {
675         StringBuffer url = new StringBuffer();
676         try {
677             url.append( _request.getURL().getProtocol() ).append( "://" );
678             url.append( _request.getURL().getHost() );
679             url.append( _request.getURL().getPath() );
680         } catch (MalformedURLException e) {
681             throw new RuntimeException( "unable to read URL from request: " + _request );
682         }
683         return url;
684     }

728     void readBasicAuthentication() {
729         String authorizationHeader = (String) _headers.get( "Authorization" );
730 
731         if (authorizationHeader != null) {
732             String userAndPassword = Base64.decode( authorizationHeader.substring( authorizationHeader.indexOf( ' ' ) + 1 ) );
733             int colonPos = userAndPassword.indexOf( ':' );
734             recordAuthenticationInfo( userAndPassword.substring( 0, colonPos ),
735                                       toArray( userAndPassword.substring( colonPos+1 ) ) );
736         }
737     }

739     static String[] toArray( String roleList ) {
740         StringTokenizer st = new StringTokenizer( roleList, "," );
741         String[] result = new String[ st.countTokens() ];
742         for (int i = 0; i < result.length; i++) {
743             result[i] = st.nextToken();
744         }
745         return result;
746     }

800         PrioritizedLocale( String languageSpec ) {
801             int semiIndex = languageSpec.indexOf( ';' );
802             if (semiIndex < 0) {
803                 _priority = 1;
804                 _locale = parseLocale( languageSpec );
805             } else {
806                 _priority = Float.parseFloat( languageSpec.substring( languageSpec.indexOf( '=', semiIndex )+1 ) );
807                 _locale = parseLocale( languageSpec.substring( 0, semiIndex ) );
808             }
809         }

812         private Locale parseLocale( String range ) {
813             range = range.trim();
814             int dashIndex = range.indexOf( '-' );
815             if (dashIndex < 0) {
816                 return new Locale( range, "" );
817             } else {
818                 return new Locale( range.substring( 0, dashIndex ), range.substring( dashIndex+1 ) );
819             }
820         }

824             return _locale;
825         }

829             if (!(o instanceof PrioritizedLocale)) throw new IllegalArgumentException( "may only combine with other prioritized locales" );
830             PrioritizedLocale other = (PrioritizedLocale) o;
