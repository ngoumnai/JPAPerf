666     /**
667      * Called by interior column tags to help this tag figure out how it is supposed to display the information in the
668      * List it is supposed to display.
669      * @param column an internal tag describing a column in this tableview
670      */
671     public void addColumn(HeaderCell column)
672     {
673         if (log.isDebugEnabled())
674         {
675             log.debug("[" + getUid() + "] addColumn " + column);
676         }
677 
678         if ((this.paginatedList != null) && (column.getSortable()))
679         {
680             String sortCriterion = paginatedList.getSortCriterion();
681 
682             String sortProperty = column.getSortProperty();
683             if (sortProperty == null)
684             {
685                 sortProperty = column.getBeanPropertyName();
686             }
687 
688             if ((sortCriterion != null) && sortCriterion.equals(sortProperty))
689             {
690                 this.tableModel.setSortedColumnNumber(this.tableModel.getNumberOfColumns());
691                 column.setAlreadySorted();
692             }
693         }
694 
695         this.tableModel.addColumnHeader(column);
696     }

698     /**
699      * Adds a cell to the current row. This method is usually called by a contained ColumnTag
700      * @param cell Cell to add to the current row
701      */
702     public void addCell(Cell cell)
703     {
704         // check if null: could be null if list is empty, we don't need to fill rows
705         if (this.currentRow != null)
706         {
707             int columnNumber = this.currentRow.getCellList().size();
708             this.currentRow.addCell(cell);
709 
710             // just be sure that the number of columns has not been altered by conditionally including column tags in
711             // different rows. This is not supported, but better avoid IndexOutOfBounds...
712             if (columnNumber < tableModel.getHeaderCellList().size())
713             {
714                 HeaderCell header = (HeaderCell) tableModel.getHeaderCellList().get(columnNumber);
715                 header.addCell(new Column(header, cell, currentRow));
716             }
717         }
718     }

720     /**
721      * Is this the first iteration?

723      */
724     protected boolean isFirstIteration()
725     {
726         if (log.isDebugEnabled())

731                 + (this.rowNumber == 1)
732                 + " (row number="
733                 + this.rowNumber

741     /**
742      * When the tag starts, we just initialize some of our variables, and do a little bit of error checking to make sure
743      * that the user is not trying to give us parameters that we don't expect.
744      * @return int
745      * @throws JspException generic exception
746      * @see javax.servlet.jsp.tagext.Tag#doStartTag()
747      */
748     public int doStartTag() throws JspException
749     {
750         DependencyChecker.check();
751 
752         // needed before column processing, elsewhere registered views will not be added
753         ExportViewFactory.getInstance();
754 
755         if (log.isDebugEnabled())
756         {
757             log.debug("[" + getUid() + "] doStartTag called");
758         }
759 
760         this.properties = TableProperties.getInstance((HttpServletRequest) pageContext.getRequest());
761         this.tableModel = new TableModel(this.properties, pageContext.getResponse().getCharacterEncoding(), pageContext);
762 
763         // copying id to the table model for logging
764         this.tableModel.setId(getUid());
765         this.tableModel.setForm(this.form);
766 
767         initParameters();
768 
769         this.tableModel.setMedia(this.currentMediaType);
770 
771         Object previousMediaType = this.pageContext.getAttribute(PAGE_ATTRIBUTE_MEDIA);
772         // set the PAGE_ATTRIBUTE_MEDIA attribute in the page scope
773         if (previousMediaType == null || MediaTypeEnum.HTML.equals(previousMediaType))
774         {
775             if (log.isDebugEnabled())
776             {
777                 log.debug("[" + getUid() + "] setting media [" + this.currentMediaType + "] in this.pageContext");
778             }
779             this.pageContext.setAttribute(PAGE_ATTRIBUTE_MEDIA, this.currentMediaType);
780         }
781 
782         doIteration();
783 
784         // always return EVAL_BODY_TAG to get column headers also if the table is empty
785         // using int to avoid deprecation error in compilation using j2ee 1.3
786         return 2;
787     }

789     /**
790      * @see javax.servlet.jsp.tagext.BodyTag#doAfterBody()
791      */
792     public int doAfterBody()
793     {
794         // doAfterBody() has been called, body is not empty
795         this.doAfterBodyExecuted = true;
796 
797         if (log.isDebugEnabled())
798         {
799             log.debug("[" + getUid() + "] doAfterBody called - iterating on row " + this.rowNumber);
800         }
801 
802         // increment this.rowNumber
803         this.rowNumber++;
804 
805         // Call doIteration() to do the common work
806         return doIteration();
807     }

809     /**
810      * Utility method that is used by both doStartTag() and doAfterBody() to perform an iteration.
811      * @return <code>int</code> either EVAL_BODY_TAG or SKIP_BODY depending on whether another iteration is desired.
812      */
813     protected int doIteration()
814     {
815 
816         if (log.isDebugEnabled())
817         {
818             log.debug("[" + getUid() + "] doIteration called");
819         }
820 
821         // Row already filled?
822         if (this.currentRow != null)
823         {
824             // if yes add to table model and remove
825             this.tableModel.addRow(this.currentRow);
826             this.currentRow = null;
827         }
828 
829         if (this.tableIterator.hasNext())
830         {
831 
832             Object iteratedObject = this.tableIterator.next();
833             if (getUid() != null)
834             {
835                 if ((iteratedObject != null))
836                 {
837                     // set object into this.pageContext
838                     if (log.isDebugEnabled())
839                     {
840                         log.debug("[" + getUid() + "] setting attribute \"" + getUid() + "\" in pageContext");
841                     }
842                     this.pageContext.setAttribute(getUid(), iteratedObject);
843 
844                 }
845                 else
846                 {
847                     // if row is null remove previous object
848                     this.pageContext.removeAttribute(getUid());
849                 }
850                 // set the current row number into this.pageContext
851                 this.pageContext.setAttribute(getUid() + TableTagExtraInfo.ROWNUM_SUFFIX, new Integer(this.rowNumber));
852             }
853 
854             // Row object for Cell values
855             this.currentRow = new Row(iteratedObject, this.rowNumber);
856 
857             this.lastIteration = !this.tableIterator.hasNext();
858 
859             // new iteration
860             // using int to avoid deprecation error in compilation using j2ee 1.3
861             return 2;
862         }
863         this.lastIteration = true;
864 
865         if (log.isDebugEnabled())
866         {
867             log.debug("[" + getUid() + "] doIteration() - iterator ended after " + (this.rowNumber - 1) + " rows");
868         }
869 
870         // end iteration
871         return SKIP_BODY;
872     }

875      * Get the given parameter from the request or, if not avaible, look for into into the session if keepstatus is set.
876      * Also takes care of storing an existing paramter into session.
877      * @param request servlet request
878      * @param requestHelper request helper instance
879      * @param parameter parameter, will be encoded
880      * @return value value taken from a request parameter or from a session attribute
881      */
882     private Integer getFromRequestOrSession(HttpServletRequest request, RequestHelper requestHelper, String parameter)
883     {
884         String encodedParam = encodeParameter(parameter);
885         Integer result = requestHelper.getIntParameter(encodedParam);
886 
887         if (keepStatus)
888         {

893                 if (session != null)
894                 {
895                     if (clearStatus)

905             else
906             {
907                 // set into session
908                 request.getSession(true).setAttribute(encodedParam, result);
909             }
910         }
911         return result;

1221     /**
1222      * Draw the table. This is where everything happens, we figure out what values we are supposed to be showing, we
1223      * figure out how we are supposed to be showing them, then we draw them.
1224      * @return int
1225      * @throws JspException generic exception
1226      * @see javax.servlet.jsp.tagext.Tag#doEndTag()
1227      */
1228     public int doEndTag() throws JspException
1229     {
1230 
1231         if (log.isDebugEnabled())
1232         {
1233             log.debug("[" + getUid() + "] doEndTag called");
1234         }
1235 
1236         if (!this.doAfterBodyExecuted)
1237         {
1238             if (log.isDebugEnabled())
1239             {
1240                 log.debug("[" + getUid() + "] tag body is empty.");
1241             }
1242 
1243             // first row (created in doStartTag)
1244             if (this.currentRow != null)
1245             {
1246                 // if yes add to table model and remove
1247                 this.tableModel.addRow(this.currentRow);
1248             }
1249 
1250             // other rows
1251             while (this.tableIterator.hasNext())
1252             {
1253                 Object iteratedObject = this.tableIterator.next();
1254                 this.rowNumber++;
1255 
1256                 // Row object for Cell values
1257                 this.currentRow = new Row(iteratedObject, this.rowNumber);
1258 
1259                 this.tableModel.addRow(this.currentRow);
1260             }
1261         }
1262 
1263         // if no rows are defined automatically get all properties from bean
1264         if (this.tableModel.isEmpty())
1265         {
1266             describeEmptyTable();
1267         }
1268 
1269         TableDecorator tableDecorator = this.properties.getDecoratorFactoryInstance().loadTableDecorator(
1270             this.pageContext,
1271             getConfiguredDecoratorName());
1272 
1273         if (tableDecorator != null)
1274         {
1275             tableDecorator.init(this.pageContext, this.list, this.tableModel);
1276             this.tableModel.setTableDecorator(tableDecorator);
1277         }
1278 
1279         setupViewableData();
1280 
1281         // Figure out how we should sort this data, typically we just sort
1282         // the data being shown, but the programmer can override this behavior
1283         if (this.paginatedList == null && this.tableModel.isLocalSort())
1284         {
1285             if (!this.tableModel.isSortFullTable())
1286             {
1287                 this.tableModel.sortPageList();
1288             }
1289         }
1290 
1291         // Get the data back in the representation that the user is after, do they want HTML/XML/CSV/EXCEL/etc...
1292         int returnValue = EVAL_PAGE;
1293 
1294         // check for nested tables
1295         // Object previousMediaType = this.pageContext.getAttribute(PAGE_ATTRIBUTE_MEDIA);
1296         Object previousMediaType = this.pageContext.getAttribute(PAGE_ATTRIBUTE_MEDIA);
1297         if (MediaTypeEnum.HTML.equals(this.currentMediaType)
1298             && (previousMediaType == null || MediaTypeEnum.HTML.equals(previousMediaType)))
1299         {
1300             writeHTMLData();
1301         }
1302         else if (!MediaTypeEnum.HTML.equals(this.currentMediaType))
1303         {
1304             if (log.isDebugEnabled())
1305             {
1306                 log.debug("[" + getUid() + "] doEndTag - exporting");
1307             }
1308 
1309             returnValue = doExport();
1310         }
1311 
1312         // do not remove media attribute! if the table is nested in other tables this is still needed
1313         // this.pageContext.removeAttribute(PAGE_ATTRIBUTE_MEDIA);
1314 
1315         if (log.isDebugEnabled())
1316         {
1317             log.debug("[" + getUid() + "] doEndTag - end");
1318         }
1319 
1320         cleanUp();
1321         return returnValue;
1322     }

1464     /**
1465      * Will write the export. The default behavior is to write directly to the response. If the ResponseOverrideFilter
1466      * is configured for this request, will instead write the exported content to a map in the Request object.
1467      * @param exportView export view
1468      * @throws JspException for problem in clearing the response or for invalid export views
1469      * @throws IOException exception thrown when writing content to the response
1470      */
1471     protected void writeExport(ExportView exportView) throws IOException, JspException
1472     {
1473         String filename = properties.getExportFileName(this.currentMediaType);
1474 
1475         HttpServletResponse response = (HttpServletResponse) this.pageContext.getResponse();
1476         HttpServletRequest request = (HttpServletRequest) this.pageContext.getRequest();
1477 
1478         Map bean = (Map) request.getAttribute(FILTER_CONTENT_OVERRIDE_BODY);
1479         boolean usingFilter = bean != null;
1480 
1481         String mimeType = exportView.getMimeType();
1482         // original encoding, be sure to add it back after reset()
1483         String characterEncoding = response.getCharacterEncoding();
1484 
1485         if (usingFilter)
1486         {
1487             if (!bean.containsKey(TableTagParameters.BEAN_BUFFER))
1488             {
1489                 // We are running under the export filter, call it
1490                 log.debug("Exportfilter enabled in unbuffered mode, setting headers");
1491                 response.addHeader(TableTagParameters.PARAMETER_EXPORTING, TagConstants.EMPTY_STRING);
1492             }
1493             else
1494             {
1495                 // We are running under the export filter in buffered mode
1496                 bean.put(TableTagParameters.BEAN_CONTENTTYPE, mimeType);
1497                 bean.put(TableTagParameters.BEAN_FILENAME, filename);
1498 
1499                 if (exportView instanceof TextExportView)
1500                 {
1501                     StringWriter writer = new StringWriter();
1502                     ((TextExportView) exportView).doExport(writer);
1503                     bean.put(TableTagParameters.BEAN_BODY, writer.toString());
1504                 }
1505                 else if (exportView instanceof BinaryExportView)
1506                 {
1507                     ByteArrayOutputStream stream = new ByteArrayOutputStream();
1508                     ((BinaryExportView) exportView).doExport(stream);
1509                     bean.put(TableTagParameters.BEAN_BODY, stream.toByteArray());
1510 
1511                 }
1512                 else
1513                 {
1514                     throw new JspTagException("Export view "
1515                         + exportView.getClass().getName()
1516                         + " must implement TextExportView or BinaryExportView");
1517                 }
1518 
1519                 return;
1520             }
1521         }
1522         else
1523         {
1524             log.debug("Exportfilter NOT enabled");
1525             // response can't be already committed at this time
1526             if (response.isCommitted())
1527             {
1528                 throw new ExportException(getClass());
1529             }
1530 
1531             try
1532             {
1533                 response.reset();
