181     private void writeFormFields()
182     {
183         Map parameters = baseHref.getParameterMap();
184 
185         ParamEncoder pe = new ParamEncoder(this.tableModel.getId());
186 
187         addIfMissing(parameters, pe.encodeParameterName(TableTagParameters.PARAMETER_ORDER));
188         addIfMissing(parameters, pe.encodeParameterName(TableTagParameters.PARAMETER_PAGE));
189         addIfMissing(parameters, pe.encodeParameterName(TableTagParameters.PARAMETER_SORT));
190 
191         for (Iterator it = parameters.keySet().iterator(); it.hasNext();)
192         {
193             String key = (String) it.next();
194             Object value = parameters.get(key);
195 
196             if (value != null & value.getClass().isArray())
197             {
198                 Object[] arr = (Object[]) value;
199                 for (int j = 0; j < arr.length; j++)
200                 {
201                     writeField(key, arr[j]);
202                 }
203             }
204             else
205             {
206                 writeField(key, value);
207             }
208         }
209     }

211     /**
212      * @param key
213      * @param value
214      */
215     private void writeField(String key, Object value)
216     {
217         StringBuffer buffer = new StringBuffer();
218         buffer.append("<input type=\"hidden\" name=\"");
219         buffer.append(esc(key));
220         buffer.append("\" value=\"");
221         buffer.append(value);
222         buffer.append("\"/>");
223 
224         write(buffer.toString());
225     }
226 
227     private String esc(Object value)
228     {
229         String valueEscaped = StringUtils.replace(ObjectUtils.toString(value), "\"", "\\\"");
230         return valueEscaped;
231     }

403     /**
404      * Writes an HTML table's column header to a JSP page.
405      * @see org.displaytag.render.TableWriterTemplate#writeTableHeader(org.displaytag.model.TableModel)
406      */
407     protected void writeTableHeader(TableModel model)
408     {
409 
410         if (log.isDebugEnabled())
411         {
412             log.debug("[" + tableModel.getId() + "] getTableHeader called");
413         }
414 
415         // open thead
416         write(TagConstants.TAG_THEAD_OPEN);
417 
418         // open tr
419         write(TagConstants.TAG_TR_OPEN);
420 
421         // no columns?
422         if (this.tableModel.isEmpty())
423         {
424             write(TagConstants.TAG_TH_OPEN);
425             write(TagConstants.TAG_TH_CLOSE);
426         }
427 
428         // iterator on columns for header
429         Iterator iterator = this.tableModel.getHeaderCellList().iterator();
430 
431         while (iterator.hasNext())
432         {
433             // get the header cell
434             HeaderCell headerCell = (HeaderCell) iterator.next();
435 
436             if (headerCell.getSortable())
437             {
438                 String cssSortable = this.properties.getCssSortable();
439                 headerCell.addHeaderClass(cssSortable);
440             }
441 
442             // if sorted add styles
443             if (headerCell.isAlreadySorted())
444             {
445                 // sorted css class
446                 headerCell.addHeaderClass(this.properties.getCssSorted());
447 
448                 // sort order css class
449                 headerCell.addHeaderClass(this.properties.getCssOrder(this.tableModel.isSortOrderAscending()));
450             }
451 
452             // append th with html attributes
453             write(headerCell.getHeaderOpenTag());
454 
455             // title
456             String header = headerCell.getTitle();
457 
458             // column is sortable, create link
459             if (headerCell.getSortable())
460             {
461                 // creates the link for sorting
462                 Anchor anchor = new Anchor(getSortingHref(headerCell), header);
463 
464                 // append to buffer
465                 header = anchor.toString();
466             }
467 
468             write(header);
469             write(headerCell.getHeaderCloseTag());
470         }
471 
472         // close tr
473         write(TagConstants.TAG_TR_CLOSE);
474 
475         // close thead
476         write(TagConstants.TAG_THEAD_CLOSE);
477 
478         if (log.isDebugEnabled())
479         {
480             log.debug("[" + tableModel.getId() + "] getTableHeader end");
481         }
482     }

484     /**
485      * Generates the link to be added to a column header for sorting.
486      * @param headerCell header cell the link should be added to
487      * @return Href for sorting
488      */
489     private Href getSortingHref(HeaderCell headerCell)
490     {
491         // costruct Href from base href, preserving parameters
492         Href href = (Href) this.baseHref.clone();
493 
494         if (this.tableModel.getForm() != null)
495         {
496             href = new PostHref(href, tableModel.getForm());
497         }
498 
499         if (this.paginatedList == null)
500         {
501             // add column number as link parameter
502             if (!this.tableModel.isLocalSort() && (headerCell.getSortName() != null))
503             {
504                 href.addParameter(encodeParameter(TableTagParameters.PARAMETER_SORT), headerCell.getSortName());
505                 href.addParameter(encodeParameter(TableTagParameters.PARAMETER_SORTUSINGNAME), "1");
506             }
507             else
508             {
509                 href.addParameter(encodeParameter(TableTagParameters.PARAMETER_SORT), headerCell.getColumnNumber());
510             }
511 
512             boolean nowOrderAscending = true;
513 
514             if (headerCell.getDefaultSortOrder() != null)
515             {
516                 boolean sortAscending = SortOrderEnum.ASCENDING.equals(headerCell.getDefaultSortOrder());
517                 nowOrderAscending = headerCell.isAlreadySorted()
518                     ? !this.tableModel.isSortOrderAscending()
519                     : sortAscending;
520             }
521             else
522             {
523                 nowOrderAscending = !(headerCell.isAlreadySorted() && this.tableModel.isSortOrderAscending());
524             }
525 
526             int sortOrderParam = nowOrderAscending ? SortOrderEnum.ASCENDING.getCode() : SortOrderEnum.DESCENDING
527                 .getCode();
528             href.addParameter(encodeParameter(TableTagParameters.PARAMETER_ORDER), sortOrderParam);
529 
530             // If user want to sort the full table I need to reset the page number.
531             // or if we aren't sorting locally we need to reset the page as well.
532             if (this.tableModel.isSortFullTable() || !this.tableModel.isLocalSort())
533             {
534                 href.addParameter(encodeParameter(TableTagParameters.PARAMETER_PAGE), 1);
535             }
536         }
537         else
538         {
539             if (properties.getPaginationSkipPageNumberInSort())
540             {
541                 href.removeParameter(properties.getPaginationPageNumberParam());
542             }
543 
544             String sortProperty = headerCell.getSortProperty();
545             if (sortProperty == null)
546             {
547                 sortProperty = headerCell.getBeanPropertyName();
548             }
549 
550             href.addParameter(properties.getPaginationSortParam(), sortProperty);
551             String dirParam;
552             if (headerCell.isAlreadySorted())
553             {
554                 dirParam = tableModel.isSortOrderAscending() ? properties.getPaginationDescValue() : properties
555                     .getPaginationAscValue();
556             }
557             else
558             {
559                 dirParam = properties.getPaginationAscValue();
560             }
561             href.addParameter(properties.getPaginationSortDirectionParam(), dirParam);
562             if (paginatedList.getSearchId() != null)
563             {
564                 href.addParameter(properties.getPaginationSearchIdParam(), paginatedList.getSearchId());
565             }
566         }
567 
568         return href;
569     }

606     /**
607      * generates the search result and navigation bar.
608      */
609     public void writeSearchResultAndNavigation()
610     {
611         if ((this.paginatedList == null && this.pagesize != 0 && this.listHelper != null)
612             || (this.paginatedList != null))
613         {
614             // create a new href
615             Href navigationHref = (Href) this.baseHref.clone();
616 
617             if (tableModel.getForm() != null)
618             {
619                 navigationHref = new PostHref(navigationHref, tableModel.getForm());
620             }
621 
622             write(this.listHelper.getSearchResultsSummary());
623 
624             String pageParameter;
625             if (paginatedList == null)
626             {
627                 pageParameter = encodeParameter(TableTagParameters.PARAMETER_PAGE);
628             }
629             else
630             {
631                 pageParameter = properties.getPaginationPageNumberParam();
632                 if ((paginatedList.getSearchId() != null)
633                     && (!navigationHref.getParameterMap().containsKey(properties.getPaginationSearchIdParam())))
634                 {
635                     navigationHref.addParameter(properties.getPaginationSearchIdParam(), paginatedList.getSearchId());
636                 }
637             }
638             write(this.listHelper.getPageNavigationBar(navigationHref, pageParameter));
639         }

642     /**
643      * Writes the formatted export links section.
644      */
645     private void writeExportLinks()
646     {
647         // Figure out what formats they want to export, make up a little string
648         Href exportHref = (Href) this.baseHref.clone();
649 
650         StringBuffer buffer = new StringBuffer(200);
651         Iterator iterator = MediaTypeEnum.iterator();
652 
653         while (iterator.hasNext())
654         {
655             MediaTypeEnum currentExportType = (MediaTypeEnum) iterator.next();
656 
657             if (this.properties.getAddExport(currentExportType))
658             {
659 
660                 if (buffer.length() > 0)
661                 {
662                     buffer.append(this.properties.getExportBannerSeparator());
663                 }
664 
665                 exportHref.addParameter(encodeParameter(TableTagParameters.PARAMETER_EXPORTTYPE), currentExportType
666                     .getCode());
667 
668                 // export marker
669                 exportHref.addParameter(TableTagParameters.PARAMETER_EXPORTING, "1");
670 
671                 Anchor anchor = new Anchor(exportHref, this.properties.getExportLabel(currentExportType));
672                 buffer.append(anchor.toString());
673             }
674         }
675 
676         String[] exportOptions = {buffer.toString()};
677         write(MessageFormat.format(this.properties.getExportBanner(), exportOptions));
678     }

680     /**
681      * create the open tag containing all the attributes.
682      * @return open tag string: <code>%lt;table attribute="value" ... ></code>
683      */
684     public String getOpenTag()
685     {
686 
687         if (this.uid != null && attributeMap.get(TagConstants.ATTRIBUTE_ID) == null)
688         {
689             // we need to clone the attribute map in order to "fix" the html id when using only the "uid" attribute
690             Map localAttributeMap = (Map) attributeMap.clone();
691             localAttributeMap.put(TagConstants.ATTRIBUTE_ID, this.uid);
692 
693             StringBuffer buffer = new StringBuffer();
694             buffer.append(TagConstants.TAG_OPEN).append(TagConstants.TABLE_TAG_NAME);
695             buffer.append(localAttributeMap);
696             buffer.append(TagConstants.TAG_CLOSE);
697 
698             return buffer.toString();
699 
700         }
701 
702         // fast, no clone
703         StringBuffer buffer = new StringBuffer();
704 
705         buffer.append(TagConstants.TAG_OPEN).append(TagConstants.TABLE_TAG_NAME);
706         buffer.append(attributeMap);
707         buffer.append(TagConstants.TAG_CLOSE);
708 
709         return buffer.toString();
710     }
